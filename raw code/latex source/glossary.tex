
%\chapter{Glossary}
%\label{chap:glossary}

%\printglossaries


\newglossaryentry{IRB}{name={IRB},
description={IRB: an institutional review board.  A committee within an institution to assess and prevent unnecessary risk to human subjects during research.}}

\newglossaryentry{BISG}{name={BISG},
description={BISG: Bayesian Improved Surname Geocoding. Using Naive Bayes to improve an estimate of an individuals ethnicity using a known last name and place of residence.}}

\newglossaryentry{Bootstrap}{name={Bootstrap},
description={Using known data as an empirical distribution for a population, and resampling the data to derive insight into aspects of the data.}}


\newglossaryentry{DOM}{name={DOM},
description={DOM: Document Object Model.  A model of an HTML document as a tree structure.}}


\newglossaryentry{CSV}{name={CSV},
description={CSV: Comma Separated Values.  A data storage format that saves data using commas to separate consecutive data.}}

\newglossaryentry{Geocoding}{name={Geocoding},
description={The process of using a database of locations to estimated a person's ethnicity using their place of residence.}}

\newglossaryentry{Pandas}{name={Pandas},
description={A Python Library for handling Panel Data}}

\newglossaryentry{Numpy}{name={Numpy},
description={Numpy: Numeric Python.  A Python Library optimized for efficient numerical analysis.}}

\newglossaryentry{P-value}{name={P-value},
description={A probability usually representing the likelihood of making an observation by chance.}}
